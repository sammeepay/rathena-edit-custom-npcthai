-	script	Mersetzdeitz#1::OutsideFlagsG5	722,{

	set .@GID, GetCastleData("gefg_cas05",1);

	if (.@GID == 0) {
		mes "[ พระราชกฤษฎีกาแห่งอาณาจักร Divine Rune-Midgarts Kingdom ]";
		mes " ";
		mes "1. ปฏิบัติตามคำสั่งของ Divine Rune-Midgarts Kingdom เราขอประกาศว่าไม่มีเจ้านายอย่างเป็นทางการของปราสาทนี้";
		mes " ";
		mes "2. สำหรับผู้ที่สามารถเอาชนะการทดสอบทั้งหมดและทำลาย Emperium ได้ กษัตริย์จะมอบกรรมสิทธิ์ในปราสาทนี้ให้กับผู้นั้น";
		close;
	}
	if (getcharid(2) == .@GID) {
		mes "[ เสียงสะท้อน ]";
		mes "ผู้กล้า...";
		mes "คุณอยากจะกลับไปยังสถานที่อันทรงเกียรติของคุณไหม?";
		next;
		if(select("กลับไปที่ปราสาทกิลด์:ออก") == 1) {
			close2;
			if (getcharid(2) == GetCastleData("gefg_cas05",1)) {
				warp "gefg_cas05",168,31;
				end;
			}
			end;
		}
		close;
	}
	mes "[ พระราชกฤษฎีกาแห่งอาณาจักร Divine Rune-Midgarts Kingdom ]";
	mes " ";
	mes "1. ปฏิบัติตามคำสั่งของ Divine Rune-Midgarts Kingdom เราอนุมัติว่าสถานที่นี้อยู่ในความครอบครองส่วนตัวของ ^ff0000"+GetGuildName(.@GID)+"^000000 Guild";
	mes " ";
	mes "2. กิลด์มาสเตอร์ของ ^ff0000"+GetGuildName(.@GID)+"^000000 กิลด์คือ ^ff0000"+GetGuildMaster(.@GID)+"^000000";
	mes "ถ้ามีใครก็ตามที่คัดค้านสิ่งนี้ จงพิสูจน์ความแข็งแกร่งและเกียรติยศของคุณด้วยดาบเหล็กในมือของคุณ";
	close;

OnRecvCastleG05:
	FlagEmblem GetCastleData("gefg_cas05",1);
	end;
}
